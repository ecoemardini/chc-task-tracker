function doGet(e) {
  try {
    var props = PropertiesService.getScriptProperties();
    var tasks = _safeLoadJSON(props, 'tasks') || [];
    var users = _safeLoadJSON(props, 'users') || [];
    var events = _safeLoadJSON(props, 'events') || [];

    return ContentService.createTextOutput(JSON.stringify({
      status: 'ok',
      tasks: tasks,
      users: users,
      events: events
    })).setMimeType(ContentService.MimeType.JSON);
  } catch (err) {
    return ContentService.createTextOutput(JSON.stringify({
      status: 'error',
      message: 'GET error: ' + err.message
    })).setMimeType(ContentService.MimeType.JSON);
  }
}

function doPost(e) {
  try {
    // Parse the incoming payload
    var raw = '';
    if (e && e.postData && e.postData.contents) {
      raw = e.postData.contents;
    } else if (e && e.postData && e.postData.getDataAsString) {
      raw = e.postData.getDataAsString();
    } else {
      return _errorResponse('No POST body received');
    }

    var payload = JSON.parse(raw);
    if (!payload || !payload.action) {
      return _errorResponse('Missing action in payload');
    }

    if (payload.action === 'fullSync') {
      return _handleSync(payload);
    }

    return _errorResponse('Unknown action: ' + payload.action);

  } catch (err) {
    return ContentService.createTextOutput(JSON.stringify({
      status: 'error',
      message: 'POST error: ' + err.message + ' | Stack: ' + (err.stack || 'none')
    })).setMimeType(ContentService.MimeType.JSON);
  }
}

function _handleSync(payload) {
  var lock = LockService.getScriptLock();
  var gotLock = false;
  try {
    gotLock = lock.tryLock(10000);
  } catch (lockErr) {
    // If lock fails, proceed without lock (rare edge case)
  }

  try {
    var props = PropertiesService.getScriptProperties();
    var serverTasks = _safeLoadJSON(props, 'tasks') || [];
    var serverUsers = _safeLoadJSON(props, 'users') || [];
    var serverEvents = _safeLoadJSON(props, 'events') || [];

    var clientTasks = payload.tasks || [];
    var clientUsers = payload.users || [];
    var clientTombstones = payload.tombstones || [];
    var clientEvents = payload.events || [];

    // Build tombstone set
    var tombstoneIds = {};
    for (var t = 0; t < clientTombstones.length; t++) {
      tombstoneIds[String(clientTombstones[t].id)] = true;
    }

    // Build server task map
    var serverMap = {};
    for (var i = 0; i < serverTasks.length; i++) {
      var sid = String(serverTasks[i].id);
      if (!tombstoneIds[sid]) {
        serverMap[sid] = serverTasks[i];
      }
    }

    // Merge client tasks (per-task, newest updatedAt wins)
    for (var j = 0; j < clientTasks.length; j++) {
      var ct = clientTasks[j];
      var cid = String(ct.id);
      if (tombstoneIds[cid]) continue;

      var existing = serverMap[cid];
      if (!existing) {
        serverMap[cid] = ct;
      } else {
        var clientTime = new Date(ct.updatedAt || ct.createdAt || '2000-01-01').getTime();
        var serverTime = new Date(existing.updatedAt || existing.createdAt || '2000-01-01').getTime();
        if (clientTime >= serverTime) {
          serverMap[cid] = ct;
        }
      }
    }

    // Convert back to array
    var mergedTasks = [];
    for (var key in serverMap) {
      if (serverMap.hasOwnProperty(key)) {
        mergedTasks.push(serverMap[key]);
      }
    }

    // Merge users
    var userMap = {};
    for (var u = 0; u < serverUsers.length; u++) {
      userMap[serverUsers[u].name] = serverUsers[u];
    }
    for (var v = 0; v < clientUsers.length; v++) {
      var cu = clientUsers[v];
      if (userMap[cu.name]) {
        if (cu.pin) userMap[cu.name].pin = String(cu.pin);
        if (cu.role) userMap[cu.name].role = cu.role;
      } else {
        userMap[cu.name] = cu;
      }
    }
    var mergedUsers = [];
    for (var ukey in userMap) {
      if (userMap.hasOwnProperty(ukey)) {
        mergedUsers.push(userMap[ukey]);
      }
    }

    // Merge events (client wins for simplicity)
    var eventMap = {};
    for (var se = 0; se < serverEvents.length; se++) {
      eventMap[serverEvents[se].id] = serverEvents[se];
    }
    for (var ce = 0; ce < clientEvents.length; ce++) {
      eventMap[clientEvents[ce].id] = clientEvents[ce];
    }
    var mergedEvents = [];
    for (var ekey in eventMap) {
      if (eventMap.hasOwnProperty(ekey)) {
        mergedEvents.push(eventMap[ekey]);
      }
    }

    // Save
    props.setProperty('tasks', JSON.stringify(mergedTasks));
    props.setProperty('users', JSON.stringify(mergedUsers));
    props.setProperty('events', JSON.stringify(mergedEvents));

    return ContentService.createTextOutput(JSON.stringify({
      status: 'ok',
      tasks: mergedTasks,
      users: mergedUsers,
      events: mergedEvents,
      lastSync: new Date().toISOString()
    })).setMimeType(ContentService.MimeType.JSON);

  } finally {
    if (gotLock) {
      try { lock.releaseLock(); } catch (e) {}
    }
  }
}

function _safeLoadJSON(props, key) {
  try {
    var raw = props.getProperty(key);
    if (!raw) return [];
    return JSON.parse(raw);
  } catch (err) {
    return [];
  }
}

function _errorResponse(msg) {
  return ContentService.createTextOutput(JSON.stringify({
    status: 'error',
    message: msg
  })).setMimeType(ContentService.MimeType.JSON);
}
