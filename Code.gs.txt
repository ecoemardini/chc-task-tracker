const SPREADSHEET_ID = ''; // Optional: set to a Google Sheet ID for backup logging
const SCRIPT_PROPS = PropertiesService.getScriptProperties();

// --- GET handler (pull) ---
function doGet(e) {
  const action = (e && e.parameter && e.parameter.action) || 'getAll';

  if (action === 'getAll') {
    const tasks = _loadTasks();
    const users = _loadUsers();
    const events = _loadEvents();
    return _jsonResponse({ status: 'ok', tasks, users, events });
  }

  return _jsonResponse({ status: 'error', message: 'Unknown action' });
}

// --- POST handler (sync) ---
function doPost(e) {
  try {
    const payload = JSON.parse(e.postData.contents);
    const action = payload.action;

    if (action === 'fullSync') {
      return _handleSync(payload);
    }

    return _jsonResponse({ status: 'error', message: 'Unknown action: ' + action });
  } catch (err) {
    return _jsonResponse({ status: 'error', message: err.message });
  }
}

// --- Sync Handler ---
function _handleSync(payload) {
  const lock = LockService.getScriptLock();
  try {
    lock.waitLock(10000); // Wait up to 10s for other syncs to finish
  } catch (e) {
    return _jsonResponse({ status: 'error', message: 'Server busy, try again' });
  }

  try {
    const serverTasks = _loadTasks();
    const serverUsers = _loadUsers();
    const serverEvents = _loadEvents();

    const clientTasks = payload.tasks || [];
    const clientUsers = payload.users || [];
    const clientTombstones = payload.tombstones || [];
    const clientEvents = payload.events || [];
    const isPartial = payload.partialSync === true;

    // Build server task map
    const serverMap = {};
    serverTasks.forEach(t => serverMap[String(t.id)] = t);

    // Apply tombstones first — remove deleted tasks from server
    const tombstoneIds = new Set(clientTombstones.map(t => String(t.id)));
    tombstoneIds.forEach(id => { delete serverMap[id]; });

    // Merge client tasks into server (per-task by updatedAt)
    clientTasks.forEach(ct => {
      const cid = String(ct.id);
      if (tombstoneIds.has(cid)) return; // skip tombstoned tasks

      const serverTask = serverMap[cid];
      if (!serverTask) {
        // New task from client
        serverMap[cid] = ct;
      } else {
        // Both exist — keep the one with the newer updatedAt
        const clientTime = new Date(ct.updatedAt || ct.createdAt || 0).getTime();
        const serverTime = new Date(serverTask.updatedAt || serverTask.createdAt || 0).getTime();

        if (clientTime >= serverTime) {
          // Client is newer or same — accept client version
          serverMap[cid] = ct;
        }
        // else: server is newer — keep server version (already in map)
      }
    });

    // Convert map back to array
    const mergedTasks = Object.values(serverMap);
    _saveTasks(mergedTasks);

    // Merge users (simple: update pins and roles)
    const serverUserMap = {};
    serverUsers.forEach(u => serverUserMap[u.name] = u);
    clientUsers.forEach(cu => {
      if (serverUserMap[cu.name]) {
        if (cu.pin) serverUserMap[cu.name].pin = String(cu.pin);
        if (cu.role) serverUserMap[cu.name].role = cu.role;
      } else {
        serverUserMap[cu.name] = cu;
      }
    });
    const mergedUsers = Object.values(serverUserMap);
    _saveUsers(mergedUsers);

    // Merge events (by ID, newest wins)
    const serverEventMap = {};
    serverEvents.forEach(e => serverEventMap[e.id] = e);
    clientEvents.forEach(ce => {
      if (!serverEventMap[ce.id]) {
        serverEventMap[ce.id] = ce;
      }
      // For events, client always wins since they're simpler
      // (no complex conflict scenarios like tasks)
      serverEventMap[ce.id] = ce;
    });
    const mergedEvents = Object.values(serverEventMap);
    _saveEvents(mergedEvents);

    return _jsonResponse({
      status: 'ok',
      tasks: mergedTasks,
      users: mergedUsers,
      events: mergedEvents
    });

  } finally {
    lock.releaseLock();
  }
}

// --- Storage helpers (using Script Properties) ---
// Script Properties has a 500KB limit per property; chunk if needed
function _loadTasks() {
  try {
    const raw = SCRIPT_PROPS.getProperty('tasks');
    return raw ? JSON.parse(raw) : [];
  } catch { return []; }
}

function _saveTasks(tasks) {
  SCRIPT_PROPS.setProperty('tasks', JSON.stringify(tasks));
}

function _loadUsers() {
  try {
    const raw = SCRIPT_PROPS.getProperty('users');
    return raw ? JSON.parse(raw) : [];
  } catch { return []; }
}

function _saveUsers(users) {
  SCRIPT_PROPS.setProperty('users', JSON.stringify(users));
}

function _loadEvents() {
  try {
    const raw = SCRIPT_PROPS.getProperty('events');
    return raw ? JSON.parse(raw) : [];
  } catch { return []; }
}

function _saveEvents(events) {
  SCRIPT_PROPS.setProperty('events', JSON.stringify(events));
}

function _jsonResponse(obj) {
  return ContentService.createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}
